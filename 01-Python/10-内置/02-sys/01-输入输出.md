```bash
python输入输出:
    1. 文件读写:
        import os
        os.write(文件描述符, "写入的内容")
    2. shell交互:
        import sys
        输入:
            sys.stdin.read(number)
        输出:
            sys.stdout.write("需要打印的内容")
    3. 打印与文件交互:
        print(value, ..., sep=" ", end="\n", file=sys.stdout, flush=False)
        print("xxx")  # xxx
        import sys
        sys.stdout = open("111.py", "a")
        print("aaa")
        print("bbb")
        import os
        os.system("cat 111.py")
        # aaa
        # bbb
```
